# import library
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# menentukan dataframe
Confirmed_Cases = c['Confirmed_Cases']
Suspected_Cases = c['Suspected_Cases']
Hospitalized = c["Hospitalized"]
Travel_History_Yes = c["Travel_History_Yes"]
Travel_History_No = c["Travel_History_No"]


# mencari mean, median dan modus

meanCC = Confirmed_Cases.mean()
medianCC = Confirmed_Cases.median()
modusCC = Confirmed_Cases.mode()

meanSC = Suspected_Cases.mean()
medianSC = Suspected_Cases.median()
modusSC = Suspected_Cases.mode()

meanH = Hospitalized.mean()
medianH = Hospitalized.median()
modusH = Hospitalized.mode()

meanTHY = Travel_History_Yes.mean()
medianTHY = Travel_History_Yes.median()
modusTHY = Travel_History_Yes.mode()

meanTHN = Travel_History_No.mean()
medianTHN = Travel_History_No.median()
modusTHN = Travel_History_No.mode()

# menampilkan ukuran nilai tengah
print("Confirmed Case -->",'Mean: ',meanCC,'Median: ',medianCC,'Modus: ',modusCC[0])
print("Susspected Case -->",'Mean: ',meanSC,'Median: ',medianSC,'Modus: ',modusSC[0])
print("Hospitalized -->",'Mean: ',meanH,'Median: ',medianH,'Modus: ',modusH[0])
print("Travel History Yes -->",'Mean: ',meanTHY,'Median: ',medianTHY,'Modus: ',modusTHY[0])
print("Travel History No -->",'Mean: ',meanTHY,'Median: ',medianTHY,'Modus: ',modusTHY[0])

#mmebuat tabel
from prettytable import PrettyTable
tabel = PrettyTable(["", "Confirmed Case", "Suspected Case", "Hospitalized", "Travel History Yes", "Travel History No"])
tabel.add_row(["Mean", meanCC, meanSC, meanH, meanTHY, meanTHN])
tabel.add_row(["Median", medianCC, medianSC, medianH, medianTHY, medianTHN])
tabel.add_row(["Modus", modusCC, modusSC, modusH, modusTHY, modusTHN])
print(tabel)
